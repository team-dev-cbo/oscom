QuartisMsgNombreReponse, Fonction

Retourne le nombre de lignes dans un T_ReponseQuartisMsg.

Syntaxe

QuartisMsgNombreReponse( RepQuartisMsg : T_ReponseQuartisMsg ) return Integer

Retour

 Nombre de réponses
-1 : Erreur T_ReponseQuartisMsg

 

Elément

Description

RepQuartisMsg

ID de la  réponse

 

 

Exemple

With stdlib ;

Procedure Quartis ;

Var

   // variables

   l_chaine : GrandeChaine ;

   l_nbMess : Integer ;

   l_res : Integer ;

   l_repQtsMsg : T_ReponseQuartisMsg ;

   l_ligneQtsMsg : T_LigneQuartisMsg ;

   l_res1 : Boolean ;

   i : Integer ;

 

Begin

 

    // Initialisation du chemin de Quartis Messenger Client

    l_chaine  := LitMemoire( "Chemin QuartisMsg" ) ;

 

    If  l_chaine <> "D:\QuartisMsg\Binaire"  Then

         l_res := QuartisMsgChemin( "D:\QuartisMsg\Binaire" ) ;

    End If ;

 

    // Lecture des messages d'une liste reçus par Quartis Messenger Client

    l_repQtsMsg := QuartisMsgLitMessage( "Liste1" ) ;

 

    // Nombre de messages reçus

    l_nbMess := QuartisMsgNombreReponse( l_repQtsMsg ) ;

 

    // Récupération des messages

    For i:=1 To l_nbMess  Do

        // Récupération d'une ligne

        l_res := QuartisMsgLigne( l_repQtsMsg , i , @l_ligneQtsMsg ) ;

 

        // Sauvegarde dans une mémoire

        If   l_res  > 0  Then

              l_chaine := l_ligneQtsMsg.QListe + " - " + l_ligneQtsMsg.QMessage + " - " +  FormaterTemps( "%d/%m/%y %H:%M:%S", l_ligneQtsMsg.QdateRecept ) ;

              l_res1 := EcritMemoireIndex("QuartisMem1" , l_chaine , i ) ;

        End If ;

 

    End For ;

 

   // Fermeture

   l_res1  := QuartisMsgFermeReponse( l_repQtsMsg ) ;

 

End  Quartis ;
