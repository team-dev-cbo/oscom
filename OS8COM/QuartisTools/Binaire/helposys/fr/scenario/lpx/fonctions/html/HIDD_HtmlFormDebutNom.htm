HtmlFormDebutNom, Fonction

 

Insère un début de formulaire nominatif dans le code source d'une page Html.

 

 

Syntaxe

 

HtmlFormDebutNom( p_action , p_nom : GrandeChaine )

 

 

Elément

Description

 

p_action

Action du formulaire

p_nom

Nom du formulaire

Exemple

 

With Html ;

 

Procedure Exemple ;

 

Var

 

l_chaine : GrandeChaine ;

 

Begin

 

 // début de la page HTML

 HtmlPageDebut( "Exemple" ,"",0 );

 

 // début du formulaire 1

 HtmlFormDebutNom( "SFAO" , "form1" );

 

 // définition d'une zone de saisie

 HtmlZoneSaisie( "zone1" , "texte1 initial" , 20 ) ;

 

 // bouton d'envoi des données

 HtmlBoutonValide( "Bouton" , "Validation" ) ;

 

 // fin du formulaire 1

 HtmlFormFin ;

 

 // ligne suivante

 HtmlLigneFin;

 

 // même formulaire aligné centre

 HtmlAligneDebut( CENTRE );

 

 // début du formulaire 2

 HtmlFormDebutNom( "SFAO" , "form2" );

 

 // définition d'une zone de saisie

 HtmlZoneSaisie( "zone1" , "texte2 initial" , 20 ) ;

 

 // bouton d'envoi des données

 HtmlBoutonValide( "Bouton" , "Validation" ) ;

 

 // fin du formulaire 2

 HtmlFormFin ;

 

 // fin de lalignement

 HtmlAligneFin;

 

 

 // même formulaire aligné droite

 HtmlAligneDebut( DROITE );

 

 // début du formulaire 3

 HtmlFormDebutNom( "SFAO" , "form3" );

 

 // définition d'une zone de saisie

 HtmlZoneSaisie( "zone1" , "texte3 initial" , 20 ) ;

 

 // bouton d'envoi des données

 HtmlBoutonValide( "Bouton" , "Validation" ) ;

 

 // fin du formulaire 3

 HtmlFormFin ;

 

 // fin de lalignement

 HtmlAligneFin;

 

 // on active la zone de saisie du 2ième formulaire

 HtmlActiveElement( "form2","zone1" );

 

 // fin de la page HTML

 HtmlPageFin;

 

 // Attente de la saisie

 AttenteSaisie ;

 

 // Lecture de la saisie

 l_chaine := CGILitValeur( "zone1" );

 

// page de réponse

 // début de la page HTML

 HtmlPageDebut( "Reponse" ,"",0 );

 

 // réponse

 HtmlTexteASCII( "Texte saisi : " + l_chaine , ROUGE );

 

 // fin de la page HTML

 HtmlPageFin;

 

 

End Exemple ;
