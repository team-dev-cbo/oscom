HtmlParagFin, Fonction

 

Insère une fin de paragraphe dans une page Html.

 

 

Syntaxe

 

HtmlParagFin

 

Exemple

 

 

WITH  Html ;

 

PROCEDURE Exemple ;

 

VAR

 l_chaine : TresGrandeChaine ;

 l_resultat : Integer ;

 

BEGIN

 

 // début de la page HTML

 HtmlPageDebut( "Exemple","",0 );

 

 // titre de la page

 HtmlTitre( "Exemple", 1,GRAS+ROUGE+ITALIQUE+CENTRE ) ;

 

 // ligne suivante

 HtmlLigneFin; 

 

 // début de paragraphe

 HtmlParagDebut;

 

 // texte

 HtmlTexte( "1ère ligne du 1er paragraphe" , 0 ) ;

 

 // ligne suivante

 HtmlLigneFin;

 

 // texte

 HtmlTexte( "2ème ligne du 1er paragraphe" , 0 ) ;

 

 // fin de paragraphe

 HtmlParagFin;

 

 // début de paragraphe

 HtmlParagDebut;

 

 // texte

 HtmlTexte( "1ère ligne du 2ème paragraphe" , 0 ) ;

 

 // ligne suivante

 HtmlLigneFin;

 

 // texte

 HtmlTexte( "2ème ligne du 2ème paragraphe" , 0 ) ;

 

 // fin de paragraphe

 HtmlParagFin;

 

 // texte

 HtmlTexte( "L'espacement entre 2 lignes d'un même paragraphe est différent de l'espacement entre 2 paragraphes" , ROUGE ) ;

 

 

 // fin de la page HTML

 HtmlPageFin;

 

END Exemple ;
