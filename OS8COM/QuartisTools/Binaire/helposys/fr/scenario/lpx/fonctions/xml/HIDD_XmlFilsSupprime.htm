XmlFilsSupprime, Fonction

 

Supprime le fils courant de l’élément courant de la structure xml.

 

 

Syntaxe

 

XmlFilsSupprime( p_xml : T_Xml ) RETURN Integer

 

Elément

Description

 

p_xml

identifiant structure xml

 

 

Valeur retournée

0 : OK ,

1 : erreur suppression,

2 : erreur identifiant structure xml.

 

 

Exemple

 

With Html, Xml ;

 

Procedure Ex3 ;

 

Var

l_chaine, l_chaine1 : GrandeChaine ;

l_res: Integer;

l_val: Integer;

l_xml,l_xml1 : T_Xml ;

 

 

Begin

 

 

// création

l_xml := XmlCree( "Xml3" );

 

// écriture du contenu d'un fichier dans une structure

l_res := XmlFichierLit( l_xml , "c:\temp\test3.txt" ) ;

 

// positionnement à la racine de la structure Xml

l_res := XmlPositionInit( l_xml ) ;

 

// positionnement sur le premier élément

l_res := XmlElementFils( l_xml ) ;

 

// lecture de sa valeur

l_res := XmlValeurLit( l_xml , @ l_chaine );

& l_chaine ;

 

// valeurs de tous les éléments

l_res := XmlElementSuivant( l_xml ) ;

While( l_res = 0 ) Do

l_res := XmlValeurLit( l_xml , @ l_chaine );

& l_chaine ;

l_res := XmlElementSuivant( l_xml ) ;

End While ;

 

 

// suppression du dernier élément en se positionnant sur la racine

l_res := XmlElementPere( l_xml ) ;

l_res := XmlFilsSupprime( l_xml ) ;

 

// positionnement sur le premier élément

l_res := XmlElementFils( l_xml ) ;

 

// lecture de sa valeur

l_res := XmlValeurLit( l_xml , @ l_chaine );

 

// valeurs de tous les éléments

l_res := XmlElementSuivant( l_xml ) ;

While( l_res = 0 ) Do

l_res := XmlValeurLit( l_xml , @ l_chaine );

& l_chaine ;

l_res := XmlElementSuivant( l_xml ) ;

End While ;

 

// fermeture

l_res := XmlFerme( l_xml ) ;

 

End Ex3 ;
