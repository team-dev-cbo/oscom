EXPORT FICHIER ASCII, Mot Clé

 

With stdlib, stdstring, database;

Procedure ExportASCII;

Var

 File : T_Fichier;

 Chaine : GrandeChaine;

 Val1 : GrandeChaine;

 Val2 : GrandeChaine;

Resultat : Boolean;

 ReponseBD : T_ReponseSQL;

 NbReponse : Integer;

 Erreur : Integer;

 NumCanal : T_CanalBD;

 

Begin

 //Ouverture du fichier ASCII

 File := OuvreFichierEcriture("C:\Import\Pointage.txt");

 

 //Réservation d'un canal pour accéder à la base de données

 NumCanal:=ReserveCanal;

 IF NumCanal.IdCanal =0 THEN

//La base de données n'est pas accessible

FermeFichier (File); //Ferme le fichier ASCII

  STOP; //Arrete l'éxécution du programme

 END IF;

 

 //Exécution de la requête sur le canal réservé

ReponseBD:=ExecuteRequeteSQLC("SELECT Champ1, Champ2 FROM Pointage WHERE Transferer = 'N'",NumCanal);

IF (ExecuteOk(ReponseBD)) THEN

NbReponse := NombreReponseSQL(ReponseBD);

ELSE

Resultat := FermeReponseSQL( ReponseBD) ;

  LibereCanal(NumCanal);

  FermeFichier(File);

  STOP;

 END IF;

 

 //Lecture du résultat

 FOR Ligne := 1 TO NbReponse DO

Val1 := ValeurChamp(ReponseBD,Ligne,1);

Val2 := ValeurChamp(ReponseBD,Ligne,2);

  Erreur := EcritureFichier(File, Val1 + Val2);

 END FOR;

 

 //Fermeture du fichier

ReponseBD:=ExecuteRequeteSQLC("UPDATE Pointage SET Transferer = '0' WHERE Transferer = 'N'",NumCanal);

 Resultat := FermeReponseSQL(ReponseBD);

 LibereCanal(NumCanal);

FermeFichier (File);

End ExportASCII;
